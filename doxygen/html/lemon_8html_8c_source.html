139 /* Next are the tables used to determine what action to take based on the
140 ** current state and lookahead token. These tables are used to implement
141 ** functions that take a state number and lookahead value and return an
142 ** action integer.
143 **
144 ** Suppose the action integer is N. Then the action is determined as
145 ** follows
146 **
147 ** 0 <= N <= YY_MAX_SHIFT Shift N. That is, push the lookahead
148 ** token onto the stack and goto state N.
149 **
150 ** N between YY_MIN_SHIFTREDUCE Shift to an arbitrary state then
151 ** and YY_MAX_SHIFTREDUCE reduce by rule N-YY_MIN_SHIFTREDUCE.
152 **
153 ** N between YY_MIN_REDUCE Reduce by rule N-YY_MIN_REDUCE
154 ** and YY_MAX_REDUCE
155 **
156 ** N == YY_ERROR_ACTION A syntax error has occurred.
157 **
158 ** N == YY_ACCEPT_ACTION The parser accepts its input.
159 **
160 ** N == YY_NO_ACTION No such action. Denotes unused
161 ** slots in the yy_action[] table.
162 **
163 ** The action table is constructed as a single large table named yy_action[].
164 ** Given state S and lookahead X, the action is computed as either:
165 **
166 ** (A) N = yy_action[ yy_shift_ofst[S] + X ]
167 ** (B) N = yy_default[S]
168 **
169 ** The (A) formula is preferred. The B formula is used instead if:
170 ** (1) The yy_shift_ofst[S]+X value is out of range, or
171 ** (2) yy_lookahead[yy_shift_ofst[S]+X] is not equal to X, or
172 ** (3) yy_shift_ofst[S] equal YY_SHIFT_USE_DFLT.

181 ** YY_SHIFT_USE_DFLT.

406 /* The following structure represents a single element of the
407 ** parser's stack. Information stored includes:
408 **
409 ** + The state number for the parser at this level of the stack.
410 **
411 ** + The value of the token stored at this level of the stack.
412 ** (In other words, the "major" token.)
413 **
414 ** + The semantic value stored at this level of the stack. This is
415 ** the information used by the action routines in the grammar.
416 ** It is sometimes called the "minor" token.
417 **
418 ** After the "shift" half of a SHIFTREDUCE action, the stateno field
419 ** actually contains the reduce action for the second half of the
420 ** SHIFTREDUCE.
421 */
422 struct yyStackEntry {
423  YYACTIONTYPE stateno; /* The state-number, or reduce action in SHIFTREDUCE */
424  YYCODETYPE major; /* The major token value. This is the code
425  ** number for the token at this stack level */
426  YYMINORTYPE minor; /* The user-supplied minor token value. This
427  ** is the value of the token */
428 };
429 typedef struct yyStackEntry yyStackEntry;

443  int yystksz; /* Current side of the stack */
444  yyStackEntry *yystack; /* The parser's stack */
445  yyStackEntry yystk0; /* First stack entry */
446 #else
447  yyStackEntry yystack[YYSTACKDEPTH]; /* The parser's stack */
448 #endif
